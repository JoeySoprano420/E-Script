E-Script EBNF Grammar

program ::=
statement*
statement ::=
operation "#"
| comment
operation ::=
"modify" identifier expression
| "adjust" identifier expression
| "bypass" identifier
| "delete" identifier
| "process" action expression?
| "create" identifier expression?
| "deploy" identifier expression?
| "lane" identifier operation
| "sync" identifier?
expression ::=
literal
| identifier
| expression "+" expression
| expression "-" expression
| "(" expression ")"
identifier ::=
[A-Za-z_][A-Za-z0-9_]*
literal ::=
string | number
string ::=
'"' [^"]* '"'
number ::=
[0-9]+ ("." [0-9]+)?
action ::=
"read" | "write" | "ping" | "sync" | "analyze" | "create"
comment ::=
"*" [^\n]* "\n"
| "**" .* "**"
